def self.create_txn(type: nil, account: nil)
  txn = self.new
  txn.account_id = account.id    
  txn.state = :initiated
  txn.type = type
  txn
end

def buy
  buy_txn = self.account.add_to_holdings(stock: self.order.stock, qty: self.order.stock_qty)
  if buy_txn
    self.state = :performed
  else
    self.state = :failed
  end
